    while(1) {
        if(xSemaphoreTake(xI2CMutex, pdMS_TO_TICKS(100))) {
            data = read_sensor_data();
            xSemaphoreGive(xI2CMutex);

            xQueueSend(xDataQueue, &data, portMAX_DELAY);
            vTaskDelay(pdMS_TO_TICKS(100));
        }
    }
